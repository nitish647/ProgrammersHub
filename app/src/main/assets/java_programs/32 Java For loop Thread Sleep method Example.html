//Simple Java For loop Thread Sleep method Example

public class Thread_first extends Thread
{
	public void run()
	{
		for(int i=1;i<=10;i++)
		{
		System.out.println(i);
		//check if exception occured
		try {
			//sleep a thread for the specified amount of time.
			Thread.sleep(1000);
			
			//catch and throw if exception occured in try block
		} catch (InterruptedException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
		}
	}
public static void main(String[] args)
{
Thread_first obj = new Thread_first();
//starts the execution of the thread.JVM calls the run() method on the thread
obj.start();
}	
}
