import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class SearchWordFromText {
	
	public static void main(String[] args) throws Exception
	{
		//If your are using eclipse then copy file.txt in your project workspace
		File file = new File("file.txt");

		try {
			//this will scan you file 
		    Scanner scanner = new Scanner(file);

		    //now read the file line by line...
		    int lineNum = 0;
		    //check until it found
		    while (scanner.hasNextLine()) {
		        String line = scanner.nextLine();
		        lineNum++;//print line number
		        	System.out.println("String is: "+line);//print found string
		            System.out.println("I found it on line " +lineNum);//print line number
		       
		    }
		} catch(FileNotFoundException e) { 
		    //handle this
			e.printStackTrace();
		}
	}
}
