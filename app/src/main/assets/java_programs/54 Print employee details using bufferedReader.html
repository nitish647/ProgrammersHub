/*
 * Print employee details by taking user input using bufferedReader
 * */

package assignments;

import java.io.BufferedReader;
import java.io.InputStreamReader;


public class Employee_details
{
    public static void main(String[] args) throws Exception
    {
        int id,salary;
        String fname,job_profile;
        BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));
        System.out.print("Enter employee id: ");
        id = Integer.parseInt(bufferedReader.readLine());
        System.out.print("Enter employee full name: ");
        fname = bufferedReader.readLine();
        System.out.print("Enter employee job profile: ");
        job_profile = bufferedReader.readLine();
        System.out.print("Enter employee salary: ");
        salary = Integer.parseInt(bufferedReader.readLine());
       
        System.out.println("********************************"); 
        System.out.println("Age: "+ id);
        System.out.println("Salary: "+ fname);
        System.out.println("Age: "+ job_profile);
        System.out.println("Salary: "+ salary);
        System.out.println("*******************************");
    }    
}
