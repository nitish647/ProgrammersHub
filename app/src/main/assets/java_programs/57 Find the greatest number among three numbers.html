/*
 * Find the greatest number among three numbers
 */
package assignments;

public class greatest_number {
    
    public static void main(String[] args)
    {
        int max = 0;
        int[] arr_list = new int[]{5,12,6};
        for(int i=1;i<=arr_list.length-1;i++)
        {
            if(arr_list[max] < arr_list[i] )
            {
               max = i;
            }
        }
        System.out.println("Largest number is: "+arr_list[max]);
    }    
}


Output:

 Largest number is: 12
