#include <iostream>
using namespace std;
class student
{
  private:
    int rollno;
    char name [20];
  public:
    void read ( )
    {
       cout <<"Enter The Roll Number & Name: "<<endl;
       cin >> rollno;
       cin.ignore();
       cin.getline(name,20);
    }
    void show ( )
    {
       cout <<"Roll No = " << rollno<<endl;
       cout <<"Name = " << name<<endl;
    }
};
class exam_internal : public student
{
   protected:
     int sub1_marks;
     int sub2_marks;
   public:
     void read_marks ( )
     {
       cout << "Enter Internal Marks Of Subject 1 = ";
       cin >> sub1_marks;
       cout << "Enter Internal Marks Of Subject 2 = ";
       cin >> sub2_marks;
     }
     void display_marks ( )
     {
       cout <<"Internal Marks Of Subject 1 = " << sub1_marks<<endl;
       cout <<"Internal Marks Of Subject 2 = " << sub2_marks<<endl;
     }
};
class exam_external
{
  protected:
    int sub1_ext_marks;
    int sub2_ext_marks;
  public:
    void read_marks ( )
    {
      cout <<"Enter External Marks of subject 1 = ";
      cin >> sub1_ext_marks;
      cout <<"Enter external marks of subject 2 = ";
      cin >> sub2_ext_marks;
    }
    void display_marks ( )
    {
      cout <<"External Marks Of Subject 1 = " << sub1_ext_marks<<endl;
      cout <<"External Marks Of Subject 2 = " << sub2_ext_marks<<endl;
    }
};
class result : public exam_internal, public exam_external
{
  private:
    int total_marks;
  public:
    void cal_result ( )
    {
      total_marks = sub1_marks + sub2_marks + sub1_ext_marks + sub2_ext_marks;
      cout << " Total Marks Obtained = " << total_marks;
    }
};
int main ( )
{
  result r1;
  cout << "Enter The Details Of Student:"<<endl;
  r1.read ( );
  cout << "Enter The Marks Of Internal Examination:"<<endl;
  r1.exam_internal :: read_marks ( );
  cout <<"Enter Marks Of External Examination: "<<endl;
  r1.exam_external :: read_marks ( );
  cout <<"Displaying Student Details: "<<endl;
  r1.show ( );
  r1.exam_internal :: display_marks ( );
  r1.exam_external :: display_marks ( );
  cout <<"Calculating & Displaying Result "<<endl;
  r1.cal_result ( );
  return 0;
}
