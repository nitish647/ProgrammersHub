#include<iostream>
using namespace std;
int main()
{
  int arr[5]={10,8,6,4,2};
  int *ptr;
  ptr=arr;
  cout<<"Array values by using pointers :"<<endl;
  for(int i=0;i<5;i++)
     cout<<*(ptr+i)<<endl;
  cout<<"Array values by using array name :"<<endl;
  for(int i=0;i<5;i++)
     cout<<*(arr+i)<<endl;
  return 0;
}

Output :
Array values by using pointers :
10
8
6
4
2
Array values by using array name :
10
8
6
4
2
