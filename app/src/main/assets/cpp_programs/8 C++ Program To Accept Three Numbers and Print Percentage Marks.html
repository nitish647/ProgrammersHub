#include <iostream>
using namespace std;
int main()
{
  float sub1, sub2, sub3, marks, perc;
  cout<<"Enter marks obtained in three subjects: ";
  cin>> sub1 >> sub2 >> sub3;
  marks = sub1 + sub2 + sub3;
  perc = (marks / 300) * 100;
  cout<<endl<<"The percentage marks are: "<<perc<<"%";
  return 0;
}

Output:
Enter marks obtained in three subjects: 90
95
92

The percentage marks are: 92.3333%
