#include <iostream>
#include<cmath>
using namespace std;
int GCD(int a, int b)
{
  int remainder;
  while( b != 0 )
  {
    remainder = a % b;
    a = b;
    b = remainder;
   }
  return a;
}
int main()
{
  int x, y;
  cout << "To find the greatest common divisor \n";
  cout << "Value 1: ";
  cin >> x;
  cout << "Value 2: ";
  cin >> y;
  cout << "\nThe Greatest Common Divisor of " << x << " and " << y << " is " << GCD(x, y) << endl;
  return 0;
}

Output:
To find the greatest common divisor
Value 1: 9
Value 2: 87

The Greatest Common Divisor of 9 and 87 is 3
