#include<iostream>
using namespace std;
class Account
{
   private:
     int balance;
   public:
     Account(int i=0)
     { balance=i; }
     void deposit(int amount)
     {
       balance += amount;
     }
     void withdraw(int amount)
     {
       balance-=amount;
     }
     int getBalance()
     {
       return balance;
     }
};
int main()
{
  Account a1(300);
  cout<<"Initial Balance : "<<a1.getBalance()<<endl;
  a1.deposit(700);
  cout<<"Balance after deposit : "<<a1.getBalance()<<endl;
  a1.withdraw(500);
  cout<<"Balance after withdraw : "<<a1.getBalance()<<endl;
}

Output :
Initial Balance : 300
Balance after deposit : 1000
Balance after withdraw : 500
