#include<iostream>
using namespace std;
class Demo
{
  public:
    Demo()
    {
      cout<<"Constructor is called!"<<endl;
    }
    ~Demo()
    {
      cout<<"Destructor is called !"<<endl;
    }
};
int main()
{
   Demo *myarray = new Demo[5];
   delete [] myarray;
   return 0;
}
