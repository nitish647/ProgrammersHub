#include <iostream>
#include <conio.h>
#include <stdio.h>
#include <math.h>
using namespace std;
class person
{
  private:
    int age;
    char name[20];
  public:
    void read()
    {
      cout<<"Enter Age & Name: ";
      cin>>age;
      cin.ignore();
      cin.getline(name,20);
    }
    void disp()
    {
      cout<<endl<<"**Details Of Person**"<<endl;
      cout<<"Age: "<<age<<endl;
      cout<<"Name: ";
      puts (name);
    }
};
class student: public person
{
  private:
     int rollno, clss, marks;
  public:
     void read()
     {
       cout<<"Enter The Roll Number, Class & Marks";
       cin>>rollno>>clss>>marks;
     }
     void disp()
     {
       cout<<"Roll Number: "<<rollno<<endl;
       cout<<"Class: "<<clss<<endl;
       cout<<"Marks: "<<marks<<endl;
     }
};
class employee: public person
{
   int id;
   char desig[25];
   char dept[20];
   public:
     void read()
     {
       cout<<"Enter The Employee ID, Department & Designation: ";
       cin>>id>>dept>>desig;
     }
     void disp()
     {
       cout<<"Employee ID: "<<id<<endl;
       cout<<"Department: "<<dept<<endl;
       cout<<"Designation: "<<desig;
     }
};
int main()
{
  student S[50];
  person P[50];
  employee E[50];
  int n, i, ch;
  cout <<"Enter Limit :";
  cin >> n;
  cout <<"***MENU***"<<endl;
  cout<<"1. Student "<<endl;
  cout<<"2. Employee "<<endl;
  cout<<"Enter Your Choice..."<<endl;
  cin >>ch;
  if (ch==1)
  {
    for (int i=0;i<n;i++)
    {
       P[i].read();
       S[i].read();
       P[i].disp();
       S[i].disp();
       cout<<endl;
    }
  }
  else if (ch==2)
  {
     for (int i=0;i<n;i++)
     {
        P[i].read();
        E[i].read();
        P[i].disp();
        E[i].disp();
        cout<<endl;
     }
  }
  else
    cout<<"Invalid Choice!";
  getch();
  return 0;
}
