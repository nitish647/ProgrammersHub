#include <iostream>
#include <iomanip>
using namespace std;
class complex
{
  private:
    double real, img;
  public:
    complex ( )
    {
      real = img = 0.0;
    }
    void read ( )
    {
      cout<<"Enter Real & Imaginary Part: ";
      cin >> real >> img;
    }
   complex operator + (complex cc2)
   {
     complex temp;
     temp.real = real + cc2.real;
     temp.img = img + cc2.img;
     return (temp);
   }
   void show ( )
   {
     cout<<"The Sum Is: "<<endl;
     if (img > 0)
       cout<<real << "+" << img << "i" << endl;
     else
       cout << real << img << "i" << endl;
   }
};
int main ( )
{
  complex c1, c2, c3;
  cout << "Enter Complex Number C1";
  c1.read ( );
  cout <<"Enter Complex Number C2";
  c2.read ( );
  c3 = c1 + c2;
  c3.show ( );
  return 0;
}
