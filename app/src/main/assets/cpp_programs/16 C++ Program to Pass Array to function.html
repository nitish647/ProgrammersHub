#include<iostream>
using namespace std;
double Average(int arr[],int size);
int main()
{
  int value[10]={12,14,3,5,6,7,20,12,10,15};
  double avg;
  avg=Average(value,10);
  cout<<"Average Value is :"<<avg<<endl;
  return 0;
}
double Average(int arr[],int size)
{
  int i,sum=0;
  double avg;
  for(i=0;i<size;i++)
    sum=sum+arr[i];
  avg=double(sum)/size;
  return avg;
}

Output :
Average Value is :10.4
