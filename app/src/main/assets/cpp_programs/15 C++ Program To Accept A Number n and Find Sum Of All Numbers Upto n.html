int main()
{
  int num = 0;
  int sumnum;
  cout <<"Enter any integer greater than 0: " << endl;
  cin >> num;
  sumnum = recursion(num);

int recursion(int num)
{
  int sum = 0;
  if(num == 1)
    return 1;
  else
  {
     sum = recursion(num - 1) + num;
     return sum;
  }
}
