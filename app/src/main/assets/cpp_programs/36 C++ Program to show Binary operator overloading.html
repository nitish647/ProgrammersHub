#include<iostream>
using namespace std;
class Cube
{
   private:
      int length,width,breadth;
   public:
      Cube()
      { }
      Cube(int l,int w,int b)
      {
        length=l;
        width=w;
        breadth=b;
      }
      int getVolume()
      {
        return length*width*breadth;
      }
      Cube operator+(Cube &c)
      {
        Cube cube;
        cube.length=this->length+c.length;
        cube.width=this->width+c.width;
        cube.breadth=this->breadth+c.breadth;
        return cube;
      }
};
int main()
{
  Cube C1(2,3,4);
  Cube C2(3,5,7),C3;
  cout<<"Volume of C1 : "<<C1.getVolume()<<endl;
  cout<<"Volume of C2 : "<<C2.getVolume()<<endl;
  C3=C1+C2;
  cout<<"Volume of C3 : "<<C3.getVolume()<<endl;
  return 0;
}

Output :
Volume of C1 : 24
Volume of C2 : 105
Volume of C3 : 440
