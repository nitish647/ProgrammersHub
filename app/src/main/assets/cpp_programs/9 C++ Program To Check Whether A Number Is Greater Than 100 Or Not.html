#include <iostream>
using namespace std;
int main()
{
  int x;
  cout << "Enter an integer: ";
  cin >> x;
  if (x < 100)
    cout <<" X is less than 100"<< endl;
  else if (x > 100)
    cout <<"X is greater than 100" << endl;
  else if (x == 100)
    cout << "X is equal to 100" << endl;
  return 0;
}
