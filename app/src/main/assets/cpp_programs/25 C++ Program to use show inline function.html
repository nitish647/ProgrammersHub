#include<iostream>
using namespace std;
inline int Max(int x,int y)
{
  return (x<y)?y:x;
}
int main()
{
  cout<<"Max (5,10): "<<Max(5,10)<<endl;
  cout<<"Max (20,10): "<<Max(20,10)<<endl;
  cout<<"Max (50,40): "<<Max(50,40)<<endl;
  return 0;
}

Output :
Max (5,10): 10
Max (20,10): 20
Max (50,40): 50
