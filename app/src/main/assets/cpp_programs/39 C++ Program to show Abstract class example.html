#include<iostream>
using namespace std;
class Shape
{
  protected:
    int width,length;
  public:
    virtual int getArea()=0;
};
class Triangle:public Shape
{
  public :
    Triangle(int l,int w)
    {
      length=l;
      width=w;
    }
    int getArea()
    {
      return (width*length)/2;
    }
};
class Rectangle:public Shape
{
   public :
     Rectangle(int l,int w)
     {
        length=l;
        width=w;
     }
     int getArea()
     {
       return (width*length);
     }
};
int main()
{
   Rectangle R(3,4);
   Triangle T(4,5);
   cout<<"Area of Rectangle R : "<<R.getArea()<<endl;
   cout<<"Area of Triangle T : "<<T.getArea()<<endl;
   return 0;
}

Output :
Area of Rectangle R : 12
Area of Triangle T : 10
