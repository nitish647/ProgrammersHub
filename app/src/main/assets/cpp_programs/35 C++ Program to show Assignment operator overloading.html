#include<iostream>
using namespace std;
class Distance
{
   private:
     int feet,inches;
   public:
     Distance()
     { feet=0;inches=0; }
     Distance(int f,int i)
     {
        feet=f;
        inches=i;
     }
     void displayDistance()
     {
        cout<<"F : "<<feet<<" L : "<<inches<<endl;
     }
     Distance operator=(Distance d)
     {
       inches=d.inches;
       feet=d.feet;
       return Distance(feet,inches);
     }
};
int main()
{
  Distance D1(3,5),D2;
  cout<<"Distance of D2 before Assignment :"<<endl;
  D2.displayDistance();
  D2=D1;
  cout<<"Distance of D2 After Assignment :"<<endl;
  D2.displayDistance();
  return 0;
}

Output :
Distance of D2 before Assignment :
F : 0 L : 0
Distance of D2 After Assignment :
F : 3 L : 5
