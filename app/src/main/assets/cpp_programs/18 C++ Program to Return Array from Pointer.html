#include<iostream>
using namespace std;
int* getArray();
int main()
{
  int *p=getArray();
  for(int i=0;i<5;i++)
  {
    cout<<"Value at *(p+"<<i<<")";
    cout<<*(p+i)<<endl;
  }
  return 0;
}
int* getArray()
{
  static int a[5];
  for(int i=0;i<5;i++)
  {
    cout<<"Enter value at index "<<i<<" :";
    cin>>a[i];
  }
  for(int i=0;i<5;i++)
    cout<<a[i]<<endl;
  return a;
}
