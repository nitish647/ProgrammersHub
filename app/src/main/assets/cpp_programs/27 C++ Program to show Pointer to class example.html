#include<iostream>
using namespace std;
class Rectangle
{
 private:
   int length,breadth;
 public :
   Rectangle(int length=5,int breadth=4)
   {
    this->length=length;
    this->breadth=breadth;
   }
   int Area()
   {
    return length*breadth;
   }
};
int main()
{
 Rectangle r1(6,4);
 Rectangle r2(5,4);
 Rectangle *rect;
 rect=&r1;
 cout<<"Area of Rectangle r1 : "<<rect->Area()<<endl;
 rect=&r2;
 cout<<"Area of Rectangle r2 : "<<rect->Area()<<endl;
 return 0;
}

Output :
Area of Rectangle r1 : 24
Area of Rectangle r2 : 20
