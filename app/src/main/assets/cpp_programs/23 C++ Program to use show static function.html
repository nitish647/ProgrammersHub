#include<iostream>
using namespace std;
class Demo
{
  private:
    int a,b;
  public:
    static int objectCount;
  Demo()
  {
    a=0;b=0;
    objectCount++;
  }
  static int getCount()
  {
    return objectCount;
  }
};
int Demo::objectCount=0;
int main()
{
  cout<<"Initially Object Count : "<<Demo::getCount()<<endl;
  Demo d1,d2,d3;
  cout<<"Final Object Count : "<<Demo::getCount()<<endl;
  return 0;
}

Output :
Initially Object Count : 0
Final Object Count : 3
