#include<iostream>
using namespace std;
class Distance
{
  private:
    int feet,inches;
  public:
    Distance()
    { }
    Distance(int f,int i)
    {
      feet=f;
      inches=i;
    }
    void displayDistance()
    {
      cout<<"F : "<<feet<<" L : "<<inches<<endl;
    }
    Distance operator-()
    {
      feet=-feet;
      inches=-inches;
      new Distance(feet,inches);
    }
    bool operator <(Distance d)
    {
      if(feet<d.feet)
        return true;
      if(feet==d.feet&&inches<d.inches)
        return true;
      return false;
    }
};
int main()
{
  Distance D1(3,5),D2(-1,4);
  if(D1<D2)
     cout<<"D1 is less than D2"<<endl;
  else
     cout<<"D2 is less than D1"<<endl;
  return 0;
}

Output :
D2 is less than D1
