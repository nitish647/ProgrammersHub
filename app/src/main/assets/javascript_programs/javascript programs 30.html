    <script>

    var now = new Date();

    var dayOfWeek = now.getDay(); // Sunday - Saturday : 0 - 6

   

    if(dayOfWeek == 5) {

        document.write("Have a nice weekend!");

    } else if(dayOfWeek == 0) {

        document.write("Have a nice Sunday!");

    } else {

        document.write("Have a nice day!");

    }

    </script>
