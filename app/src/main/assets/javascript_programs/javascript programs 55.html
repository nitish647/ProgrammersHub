<body>

    <script>

    // An array with some elements

    var fruits = ["Apple", "Banana", "Mango", "Orange", "Papaya"];

    // Loop through all the elements in the array
for(var i in fruits) { 

        document.write("<p>" + fruits[i] + "</p>");
}
</script>
